#!/bin/bash

# ============================================================
# Plan Check — Agency V2 Autonomous Build Script
# Run with: bash build-agency-v2.sh
# Requires: Claude Code installed (claude command available)
# ============================================================

set -e  # Stop immediately if any step fails

# Colours for terminal output
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
RED='\033[0;31m'
NC='\033[0m' # No colour

log() { echo -e "${GREEN}[$(date +%H:%M:%S)] ✓ $1${NC}"; }
warn() { echo -e "${YELLOW}[$(date +%H:%M:%S)] ⚠ $1${NC}"; }
fail() { echo -e "${RED}[$(date +%H:%M:%S)] ✗ $1${NC}"; exit 1; }

# ------------------------------------------------------------
# Pre-flight check
# ------------------------------------------------------------

log "Starting Agency V2 build..."
log "Checking claude command is available..."
command -v claude >/dev/null 2>&1 || fail "Claude Code not found. Install from: https://claude.ai/code"

log "Checking we are in the right directory..."
[ -f "package.json" ] || fail "No package.json found. Run this script from the root of your Plan Check project."

log "Checking git status is clean..."
if [ -n "$(git status --porcelain)" ]; then
  warn "You have uncommitted changes. Committing them before we start..."
  git add -A
  git commit -m "chore: save work before agency-v2 build"
fi

log "All pre-flight checks passed. Starting build phases..."
echo ""

# PHASE 1 — Scaffold parallel route
# ============================================================

The tech stack is Next.js (App Router), Supabase, TypeScript, Tailwind, Shadcn.

TASK: Create a new parallel route at app/agency-v2/page.tsx.

Instructions:

4. Do not change any logic, data fetching, or layout yet.
5. Make sure the new route compiles without errors — check for any import path issues caused by the new location and fix them.

This is purely a safe duplication step. No functional changes.
"

echo ""
